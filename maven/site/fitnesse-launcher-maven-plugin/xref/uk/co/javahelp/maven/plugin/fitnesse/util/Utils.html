29      public static String getRelativePath(final File from, final File to) throws IOException {
30      	final String canonicalFrom = from.getCanonicalPath();
31      	final String canonicalTo = to.getCanonicalPath();
32      	if(canonicalFrom.equals(canonicalTo)) {
33      		return ".";
34      	}
35      	final String[] partsFrom = canonicalFrom.split("/");
36      	final String[] partsTo = canonicalTo.split("/");
37      	if(partsFrom.length == 0 || partsTo.length == 0) {
38      		return canonicalTo;
39      	}
40      	final List<String> listFrom = new ArrayList<String>(Arrays.asList(partsFrom));
41      	final List<String> listTo = new ArrayList<String>(Arrays.asList(partsTo));
42      	
43      	boolean same = true;
44      	int i = 0;
45      	while(same && i < partsFrom.length && i < partsTo.length) {
46      		same = partsFrom[i].equals(partsTo[i]);
47      		if(same) {
48      			listFrom.remove(0);
49      			listTo.remove(0);
50      			i++;
51      		}
52      	}
53      	
54      	final StringBuilder relativePath = new StringBuilder();
55      	final int fromSize = listFrom.size();
56      	if(fromSize == partsFrom.length - 1) {
57         	    relativePath.append("/");
58      	} else {
59      	    for(int j = 0 ; j < fromSize ; j++ ) {
60          		relativePath.append("../");
61      		}
62      	}
63      	for(final String part : listTo) {
64          	relativePath.append(part);
65          	relativePath.append("/");
66      	}
67      	relativePath.deleteCharAt(relativePath.length() - 1);
68      	return relativePath.toString();
69      }

82  	/**
83  	 * http://stackoverflow.com/questions/4520216/how-to-add-test-coverage-to-a-private-constructor
84  	 */
85  	public static <C> void assertUtilityClass(Class<C> clazz)
86  			throws InstantiationException, IllegalAccessException, InvocationTargetException {
87  		assertTrue("Utility class should be 'final'", Modifier.isFinal(clazz.getModifiers()));
88  		assertTrue("Utility class should be 'public'", Modifier.isPublic(clazz.getModifiers()));
89  		Constructor<?>[] ctors = clazz.getDeclaredConstructors();
90  		assertEquals("Utility class should only have one constructor", 1, ctors.length);
91  		Constructor<?> ctor = ctors[0];
92  		assertFalse("Utility class constructor should be inaccessible", ctor.isAccessible());
93  		ctor.setAccessible(true); // obviously we'd never do this in production
94  		assertEquals("You'd expect the construct to return the expected type", clazz, ctor.newInstance().getClass());
95  	}
